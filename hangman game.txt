import random

def hangman_game():
    # Let's keep it simple—five words to guess from
    words = ["python", "hangman", "programmer", "logic", "smiley"]
    # Pick a word at random
    secret_word = random.choice(words)
    # Set up the game state
    guesses_left = 6
    guessed_letters = []
    # Prepare the word display (underscores for unguessed letters)
    display = ["_" for _ in secret_word]

    print("\nWelcome to Hangman! You get 6 wrong guesses. Good luck!")

    # Main game loop
    while True:
        # Show current word state and letters guessed
        print("\nWord: ", " ".join(display))
        print("Guessed letters: ", " ".join(guessed_letters))

        # Get the player's guess
        guess = input("\nGuess a letter: ").lower()

        # Simple validation—just one letter, please
        if len(guess) != 1 or not guess.isalpha():
            print("Hey, please guess a single letter (a-z).")
            continue
        if guess in guessed_letters:
            print("You already guessed that letter. Try another!")
            continue

        # Add the guess to the history
        guessed_letters.append(guess)

        # Check if the guess is correct
        if guess in secret_word:
            print("Nice one!")
            # Update the display
            for i in range(len(secret_word)):
                if secret_word[i] == guess:
                    display[i] = guess
        else:
            guesses_left -= 1
            print(f"Nope! You have {guesses_left} guesses left.")

        # Check for a win or loss
        if "_" not in display:
            print("\nYou win! The word was:", secret_word)
            break
        if guesses_left <= 0:
            print("\nGame over! The secret word was:", secret_word)
            break

# Let's play!
hangman_game()
